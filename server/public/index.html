send-alert.ts 
const response = await fetch('https://api.pingui.com/alert', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_SECRET_KEY'
  },
  body: JSON.stringify({
    alert: '🚀 Deployment successful!'
  })
});
